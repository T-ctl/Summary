            const total = WARD_CONFIG[ward] || 0;
            const occupied = patients.filter(p => p.ward === ward && p.status !== 'Отпущен').map(p => p.bed);
            const free = [];
            for (let i=1; i<=total; i++) if (!occupied.includes(i)) free.push(i);
            return free;
        }
        function getNextReserveBed() {
            const reservePatients = patients.filter(p => p.ward === 'Резерв');
            return reservePatients.length === 0 ? 1 : Math.max(...reservePatients.map(p => p.bed)) + 1;
        }

        function filterDiagnoses(query) {
            if (!query) {
                const common = COMMON_DIAGS.map(d => DIAGNOSES.find(x => x.порционник === d)).filter(Boolean);
                const others = DIAGNOSES.filter(d => !COMMON_DIAGS.includes(d.порционник));
                return [...common, ...others].slice(0, 8);
            }
            const lower = query.toLowerCase();
            return DIAGNOSES.filter(d => d.порционник.toLowerCase().includes(lower)).slice(0, 8);
        }

        // Действия
        function actionFromModal(action) {
            const id = document.getElementById('edit-id').value;
            if (!id) return;
            const p = patients.find(x => x.id == id);
            if (!p) return;
            const now = new Date();
            const dateStr = now.toLocaleDateString('ru-RU');

            if (action === 'release') {
                if (!confirm('Отпустить пациента (перевести в резерв)?')) return;
                p.ward = 'Резерв';
                p.bed = getNextReserveBed();
                p.status = 'Отпущен';
                eventLog.push({ type:'release', patientId:p.id, fio:p.fio, date: dateStr, time: now.toLocaleTimeString('ru-RU', {hour:'2-digit', minute:'2-digit'}) });
                save(); closeEditModal(); renderAll();
            } else if (action === 'discharge') {
                let dischargeTime = prompt('Введите время выписки (ЧЧ:ММ)', '15:00');
                if (dischargeTime === null) return;
                if (!/^\d{2}:\d{2}$/.test(dischargeTime)) { alert('Неверный формат времени'); return; }
                if (!confirm(`Выписать пациента в ${dischargeTime}?`)) return;
                discharged.push({...p, dischargeDate: dateStr, dischargeTime: dischargeTime});
                patients = patients.filter(x=>x.id!=id);
                eventLog.push({ type:'discharge', patientId:p.id, fio:p.fio, diagnosis:p.diagnosis, date: dateStr, time: dischargeTime });
                save(); closeEditModal(); renderAll();
            } else if (action === 'deceased') {
                let deathTime = prompt('Введите время смерти (ЧЧ:ММ)', now.toTimeString().slice(0,5));
                if (deathTime === null) return;
                if (!/^\d{2}:\d{2}$/.test(deathTime)) { alert('Неверный формат времени'); return; }
                if (!confirm(`Отметить умершим в ${deathTime}?`)) return;
                eventLog.push({ type:'deceased', patientId:p.id, fio:p.fio, date: dateStr, time: deathTime });
                patients = patients.filter(x=>x.id!=id);
                save(); closeEditModal(); renderAll();
            }
        }

        // Сохранение пациента
        function savePatient() {
            const id = document.getElementById('edit-id').value;
            const ward = document.getElementById('edit-ward').value;
            const bed = parseInt(document.getElementById('edit-bed').value);
            const fio = document.getElementById('edit-fio').value.trim();
            const age = parseInt(document.getElementById('edit-age').value);
            const gender = document.getElementById('edit-gender').value;
            const ib = document.getElementById('edit-ib').value.trim();
            const diet = document.getElementById('edit-diet').value;
            const payment = document.getElementById('edit-payment').value;
            const admissionDate = document.getElementById('edit-admission-date').value;
            const admissionTime = document.getElementById('edit-admission-time').value;
            const diagnosis = document.getElementById('edit-diagnosis').value.trim() || 'Не указан';
            const mkb = document.getElementById('edit-mkb').value.trim();
            let pregnant = document.getElementById('edit-pregnant').value.trim();
            let postpartum = document.getElementById('edit-postpartum').checked;
            if (gender === 'М') { pregnant = ''; postpartum = false; }
            let severity = document.getElementById('edit-severity').value;
            const covid = document.getElementById('edit-covid').checked;
            const hiv = document.getElementById('edit-hiv').checked;
            const outOfRegion = document.getElementById('edit-out-of-region').checked;
            let oxygenLiters = parseFloat(document.getElementById('edit-oxygen-liters').value);
            if (isNaN(oxygenLiters) || oxygenLiters <= 0) oxygenLiters = 0;
            if (oxygenLiters > 5) severity = 'тяжелая';

            if (!fio || !ib || isNaN(age) || isNaN(bed)) { alert('Заполните основные поля'); return; }

            if (id) {
                const p = patients.find(x => x.id == id);
                if (!p) return;
                const oldDiag = p.diagnosis;
                Object.assign(p, { fio, age, gender, ib, diet, payment, admissionDate, admissionTime, diagnosis, mkb, pregnant, severity, postpartum, covid, hiv, outOfRegion, oxygenLiters });
                if (oldDiag !== diagnosis) {
                    eventLog.push({ type:'diagnosis_change', patientId:p.id, fio, oldDiagnosis:oldDiag, newDiagnosis:diagnosis, date: new Date().toLocaleDateString('ru-RU'), time: new Date().toLocaleTimeString('ru-RU', {hour:'2-digit', minute:'2-digit'}) });
                }
            } else {
                if (!ward || !bed) { alert('Выберите палату и койку'); return; }
                const existing = patients.find(p => p.ward === ward && p.bed === bed);
                if (existing && existing.status !== 'Отпущен') { alert('Койка занята'); return; }
                if (existing && existing.status === 'Отпущен') {
                    existing.ward = 'Резерв';
                    existing.bed = getNextReserveBed();
                }
                const newPatient = {
                    id: nextId++, ward, bed, fio, age, gender, ib, diet, payment, admissionDate, admissionTime, diagnosis, mkb,
                    pregnant, severity, postpartum, covid, hiv, outOfRegion, oxygenLiters, status: 'В палате'
                };
                patients.push(newPatient);
                eventLog.push({ type:'admission', patientId:newPatient.id, fio, ward, bed, diagnosis, date: admissionDate, time: admissionTime });
            }
            save(); closeEditModal(); renderAll();
        }

        // Отмена действий
        function openUndoModal() {
            const list = document.getElementById('event-log-list');
            list.innerHTML = '';
            if (eventLog.length === 0) { list.innerHTML = '<p>Журнал действий пуст.</p>'; }
            else {
                [...eventLog].reverse().forEach((entry, index) => {
                    const div = document.createElement('div'); div.className = 'log-entry';
                    let desc = '';
                    switch (entry.type) {
                        case 'admission': desc = `Поступление: ${entry.fio} в палату ${entry.ward}, койка ${entry.bed} (${entry.date} ${entry.time})`; break;
                        case 'transfer_internal': desc = `Перевод внутри: ${entry.fio} из ${entry.fromWard} в ${entry.toWard} (${entry.date} ${entry.time})`; break;
                        case 'transfer_out': desc = `Перевод в другое отд.: ${entry.fio} → ${entry.toDept} (${entry.date} ${entry.time})`; break;
                        case 'release': desc = `Отпуск: ${entry.fio} (${entry.date} ${entry.time})`; break;
                        case 'return_from_release': desc = `Возврат из отпуска: ${entry.fio} в палату ${entry.ward}, койка ${entry.bed} (${entry.date} ${entry.time})`; break;
                        case 'discharge': desc = `Выписка: ${entry.fio} (${entry.date} ${entry.time})`; break;
                        case 'deceased': desc = `Смерть: ${entry.fio} (${entry.date} ${entry.time})`; break;
                        case 'diagnosis_change': desc = `Смена диагноза: ${entry.fio} с "${entry.oldDiagnosis}" на "${entry.newDiagnosis}" (${entry.date} ${entry.time})`; break;
                        default: desc = JSON.stringify(entry);
                    }
                    div.innerHTML = `<span>${desc}</span> <button class="btn btn-sm btn-danger" onclick="undoAction(${eventLog.length - 1 - index})">Отменить</button>`;
                    list.appendChild(div);
                });
            }
            document.getElementById('undoModal').style.display = 'block';
        }

        function undoAction(logIndex) {
            if (!confirm('Отменить это действие? Это может затронуть другие операции.')) return;
            const entry = eventLog[logIndex];
            eventLog.splice(logIndex, 1);
            switch (entry.type) {
                case 'admission':
                    patients = patients.filter(p => p.id !== entry.patientId);
                    break;
                case 'transfer_internal': {
                    const p = patients.find(x => x.id === entry.patientId);
                    if (p) { p.ward = entry.fromWard; p.bed = entry.fromBed || 1; }
                    break;
                }
                case 'transfer_out':
                    alert('Для отмены перевода в другое отделение восстановите пациента вручную.');
                    break;
                case 'release': {
                    const p = patients.find(x => x.id === entry.patientId);
                    if (p) {
                        p.status = 'В палате';
                        const free = getFreeBeds(p.ward);
                        if (free.length > 0) p.bed = free[0];
                        else {
                            const corridorFree = getFreeBeds('Коридор');
                            if (corridorFree.length > 0) { p.ward = 'Коридор'; p.bed = corridorFree[0]; }
                        }
                    }
                    break;
                }
                case 'return_from_release': {
                    const p = patients.find(x => x.id === entry.patientId);
                    if (p) { p.status = 'Отпущен'; p.ward = 'Резерв'; }
                    break;
                }
                case 'discharge': {
                    const disPatient = discharged.find(p => p.id === entry.patientId);
                    if (disPatient) {
                        discharged = discharged.filter(p => p.id !== entry.patientId);
                        patients.push(disPatient);
                    }
                    break;
                }
                case 'deceased':
                    alert('Восстановление удалённого пациента временно невозможно.');
                    break;
                case 'diagnosis_change': {
                    const p = patients.find(x => x.id === entry.patientId);
                    if (p) p.diagnosis = entry.oldDiagnosis;
                    break;
                }
            }
            save();
            document.getElementById('undoModal').style.display = 'none';
            renderAll();
        }

        function changeCorridorBeds(delta) {
            const newCount = WARD_CONFIG['Коридор'] + delta;
            if (newCount < 0 || newCount > 6) return;
            if (delta < 0) {
                for (let b = newCount+1; b <= WARD_CONFIG['Коридор']; b++)
                    if (patients.some(p => p.ward === 'Коридор' && p.bed === b && p.status !== 'Отпущен')) {
                        alert('На удаляемых койках есть пациенты!'); return;
                    }
            }
            WARD_CONFIG['Коридор'] = newCount;
            save(); renderAll();
        }

        // Рендер таблицы
        function renderAll() {
            const tbody = document.getElementById('portion-table-body');
            const search = document.getElementById('search-input').value.trim().toLowerCase();
            tbody.innerHTML = '';

            function renderWard(ward, totalBeds) {
                const rowsData = [];
                for (let bed = 1; bed <= totalBeds; bed++) {
                    const p = patients.find(pat => pat.ward === ward && pat.bed === bed);
                    rowsData.push({ bed, patient: p });
                }
                const filtered = search ? rowsData.filter(r => r.patient && r.patient.fio.toLowerCase().includes(search)) : rowsData;
                if (search && filtered.length === 0) return;

                const trList = [];
                filtered.forEach((r) => {
                    const tr = document.createElement('tr');
                    const isFree = !r.patient;
                    const isHome = r.patient && r.patient.status === 'Отпущен';
                    let rowClass = isFree ? 'free-row' : (isHome ? 'released-row' : 'occupied-row');
                    if (r.patient && r.patient.outOfRegion) rowClass += ' out-of-region-row';
                    tr.className = rowClass;
                    tr.dataset.ward = ward;

                    tr.addEventListener('mouseenter', function() {
                        this.classList.add('highlight');
                        const wardCell = tbody.querySelector(`.ward-cell[data-ward="${ward}"]`);
                        if (wardCell) wardCell.classList.add('highlight');
                    });
                    tr.addEventListener('mouseleave', function() {
                        this.classList.remove('highlight');
                        const wardCell = tbody.querySelector(`.ward-cell[data-ward="${ward}"]`);
                        if (wardCell) wardCell.classList.remove('highlight');
                    });

                    tr.onclick = function(e) {
                        if (e.target.tagName === 'BUTTON' || e.target.tagName === 'SELECT' || e.target.tagName === 'INPUT') return;
                        if (isFree) openAddModal(ward, r.bed);
                        else openEditModal(r.patient.id);
                    };

                    let marks = [];
                    let dateStr = '';
                    if (!isFree) {
                        if (r.patient.admissionDate) {
                            const parts = r.patient.admissionDate.split('-');
                            if (parts.length === 3) dateStr = parts[2] + '.' + parts[1];
                        }
                        if (r.patient.pregnant) marks.push(`Берем. ${r.patient.pregnant}`);
                        if (r.patient.postpartum) marks.push('Послерод.');
                        if (r.patient.severity === 'тяжелая') marks.push('Тяжелая');
                        if (r.patient.covid) marks.push('COVID-19');
                        if (r.patient.hiv) marks.push('Ф-50');
                        if (r.patient.oxygenLiters > 0) marks.push(`О2-${r.patient.oxygenLiters}л`);
                    }

                    tr.innerHTML = `
                        <td>${r.bed}</td>
                        <td>${isFree ? '— свободно —' : r.patient.fio}</td>
                        <td>${isFree ? '' : r.patient.age}</td>
                        <td>${isFree ? '' : r.patient.gender}</td>
                        <td>${isFree ? '' : r.patient.ib}</td>
                        <td>${isFree ? '' : (r.patient.diet || '')}</td>
                        <td>${isFree ? '' : r.patient.payment}</td>
                        <td>${isFree ? '' : dateStr}</td>
                        <td>${isFree ? '' : r.patient.diagnosis}</td>
                        <td>${isFree ? '' : (r.patient.mkb || '')}</td>
                        <td>${isFree ? '' : (marks.join(', ') || '—')}</td>
                    `;
                    trList.push(tr);
                });

                if (trList.length === 0) return;

                const firstTr = trList[0];
                const tdWard = document.createElement('td');
                tdWard.rowSpan = trList.length;
                tdWard.className = 'ward-cell';
                tdWard.dataset.ward = ward;
                if (ward === 'Коридор') {
                    tdWard.innerHTML = `Коридор <span class="ward-controls"><button onclick="event.stopPropagation(); changeCorridorBeds(-1)">−</button><button onclick="event.stopPropagation(); changeCorridorBeds(1)">+</button></span>`;
                } else {
                    tdWard.textContent = ward;
                }
                if (wardInfectious[ward]) { tdWard.classList.add('ward-infectious'); tdWard.textContent += ' ⚠'; }
                tdWard.onclick = function(e) { e.stopPropagation(); toggleInfectious(ward); };
                firstTr.insertBefore(tdWard, firstTr.firstChild);

                trList.forEach(tr => tbody.appendChild(tr));
            }

            WARD_ORDER.forEach(ward => { if (ward !== 'Резерв') renderWard(ward, WARD_CONFIG[ward]); });

            // Резерв – только занятые койки, без пустых
            const reservePatients = patients.filter(p => p.ward === 'Резерв').sort((a, b) => a.bed - b.bed);
            if (reservePatients.length > 0) {
                const trList = [];
                reservePatients.forEach(p => {
                    const tr = document.createElement('tr');
                    tr.className = 'released-row';
                    if (p.outOfRegion) tr.classList.add('out-of-region-row');
                    tr.dataset.ward = 'Резерв';
                    tr.addEventListener('mouseenter', function() {
                        this.classList.add('highlight');
                        const wardCell = tbody.querySelector('.ward-cell[data-ward="Резерв"]');
                        if (wardCell) wardCell.classList.add('highlight');
                    });
                    tr.addEventListener('mouseleave', function() {
                        this.classList.remove('highlight');
                        const wardCell = tbody.querySelector('.ward-cell[data-ward="Резерв"]');
                        if (wardCell) wardCell.classList.remove('highlight');
                    });
                    tr.onclick = () => openEditModal(p.id);

                    let marks = [];
                    let dateStr = '';
                    if (p.admissionDate) {
                        const parts = p.admissionDate.split('-');
                        if (parts.length === 3) dateStr = parts[2] + '.' + parts[1];
                    }
                    if (p.pregnant) marks.push(`Берем. ${p.pregnant}`);
                    if (p.postpartum) marks.push('Послерод.');
                    if (p.severity === 'тяжелая') marks.push('Тяжелая');
                    if (p.covid) marks.push('COVID-19');
                    if (p.hiv) marks.push('Ф-50');
                    if (p.oxygenLiters > 0) marks.push(`О2-${p.oxygenLiters}л`);

                    tr.innerHTML = `
                        <td>${p.bed}</td>
                        <td>${p.fio}</td>
                        <td>${p.age}</td>
                        <td>${p.gender}</td>
                        <td>${p.ib}</td>
                        <td>${p.diet || ''}</td>
                        <td>${p.payment}</td>
                        <td>${dateStr}</td>
                        <td>${p.diagnosis}</td>
                        <td>${p.mkb || ''}</td>
                        <td>${marks.join(', ') || '—'}</td>
                    `;
                    trList.push(tr);
                });

                const firstTr = trList[0];
                const tdWard = document.createElement('td');
                tdWard.rowSpan = trList.length;
                tdWard.className = 'ward-cell';
                tdWard.dataset.ward = 'Резерв';
                tdWard.textContent = 'Резерв';
                tdWard.onclick = function(e) { e.stopPropagation(); /* заразность для резерва не имеет смысла */ };
                firstTr.insertBefore(tdWard, firstTr.firstChild);

                trList.forEach(tr => tbody.appendChild(tr));
            }
        }

        function updatePrintDate() {
            const now = new Date();
            const day = String(now.getDate()).padStart(2, '0');
            const month = String(now.getMonth() + 1).padStart(2, '0');
            const year = now.getFullYear();
            document.getElementById('print-date').textContent = `на ${day}.${month}.${year}`;
        }

        window.onload = function() {
            document.getElementById('display-date').textContent = new Date().toLocaleDateString('ru-RU');
            updatePrintDate();
            load();
            renderAll();
        };
